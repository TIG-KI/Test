#include <stdio.h>
main()
{
  int n,m;
  while(scanf("%d%d",&n,&m)!=EOF)
  {
    if(m<13)
    {
      if(n%m!=0)
         printf("%d\r\n",n/m+1);
      else
          printf("%d\r\n",n/m);
    }
    else if(m==13)
    {
      if(n%12!=0)
         printf("%d\r\n",n/12+1);
      else
          printf("%d\r\n",n/12);
    }
    else
    {
       if(n%m!=13)
       {
           if(n%m==0)
               printf("%d\r\n",n/m);
           else
               printf("%d\r\n",n/m+1);
       }
       else
       {
           while(n%m==13&&m>=13)
           {
             m--;
           }
           if(m==13)
           {
             if(n%12!=0)
                printf("%d\r\n",n/12+1);
             else
                printf("%d\r\n",n/12);
           }
           else
           {
             if(n%m==0)
                printf("%d\r\n",n/m);
             else
                printf("%d\r\n",n/m+1);
           }
       }
    }
  }
}
